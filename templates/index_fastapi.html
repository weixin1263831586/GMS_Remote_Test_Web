完整脚本示例

                    

                        #!/bin/bash
# 自动化测试脚本示例

BASE_URL="http://172.16.14.233:5001"
DEVICE_ID="RK3562GMS1"
TEST_TYPE="CTS"
TEST_MODULE="CtsKeystoreTestCases"
TEST_CASE="android.keystore.cts.KeyAttestationTest#testRsaAttestation"
TEST_SUITE="/home/hcq/GMS-Suite/android-cts-16_r4/android-cts/tools"

# 步骤1: 获取客户端信息
echo "步骤1: 获取客户端信息..."
curl -s "$BASE_URL/api/users/current" | jq "."

# 步骤2: 检查设备状态
echo "步骤2: 检查设备状态..."
curl -s "$BASE_URL/api/devices/list" | jq "."

# 步骤3: 启动测试（自动锁定设备）
echo "步骤3: 启动测试（会自动锁定设备）..."
START_RESULT=$(curl -s -X POST "$BASE_URL/api/test/start" \
  -H "Content-Type: application/json" \
  -d "{
    \"devices\": [\"$DEVICE_ID\"],
    \"test_type\": \"$TEST_TYPE\",
    \"test_module\": \"$TEST_MODULE\",
    \"test_case\": \"$TEST_CASE\",
    \"test_suite\": \"$TEST_SUITE\"
  }")

echo "$START_RESULT" | jq "."

# 检查是否启动成功
if echo "$START_RESULT" | jq -e '.success' > /dev/null; then
    echo "✅ 测试已启动"

    # 步骤4: 轮询测试状态
    echo "步骤4: 等待测试完成..."
    while true; do
        STATUS=$(curl -s "$BASE_URL/api/test/status")
        RUNNING=$(echo "$STATUS" | jq -r '.running')

        if [ "$RUNNING" = "false" ]; then
            echo "✅ 测试已完成"
            break
        fi

        echo "⏳ 测试运行中... $(date '+%H:%M:%S')"
        sleep 10
    done

    # 步骤5: 查看结果
    echo "步骤5: 查看测试结果..."
    curl -s "$BASE_URL/api/reports/list" | jq ".reports[0]"
else
    echo "❌ 测试启动失败"
    exit 1
fi
